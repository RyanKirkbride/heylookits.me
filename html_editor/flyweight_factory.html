import java.util.HashMap;

public class FlyweightFactory {

	private HashMap flyweightList;

	
	public FlyweightFactory()
	{
		flyweightList = new HashMap();
	}

	public Flyweight getFlyweight(Character key) {
		
		if(flyweightList.get(key) != null)
		{
			return flyweightList.get(key);
		}
		else
		{
			Flyweight conc = new Flyweight(key);
			flyweightList.put(key, conc);
			return conc;
		}
	
	}
}
