public class ListIterator implements Iterator {
	
	
	List list;
	int index = 0;
		
	public ListIterator(List l)
	{
		list = l;
	}
	
	
	@Override
	public void first() {
		index = 0;
	}

	@Override
	public void next() {
		index++;
	}

	@Override
	public boolean isDone() {
		return index >= list.count();
	}

	@Override
	public Component currentItem() {
		return (Component) list.getAt(index);
	}

}
